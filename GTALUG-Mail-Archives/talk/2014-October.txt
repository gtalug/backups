  xmodmap /home/pking/.Xmodmap
  xset +fp /usr/share/fonts/corefonts 
  xset +fp /usr/share/fonts/default 
  xset +fp /usr/share/fonts/ttf-bitstream-vera

  xset +fp /usr/share/fonts/proggy-fonts
  xset +fp /usr/share/fonts/terminus
  xset +fp /usr/share/fonts/TTF
  # for continuous landscape + landscape:
  xrandr --output DVI-I-2 --right-of DVI-I-1 
  # for portrait + landscape:
  xrandr --output DVI-I-1 --rotate left
  xsetroot -cursor_name left_ptr
